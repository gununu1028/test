問題3def print_temperature(temperature, formatter):    print(formatter(temperature))def with_celsius(temperature):    return f"温度は{temperature}℃です"def with_fahrenheit(temperature):    f = temperature * 9 / 5 + 32    return f"温度は{f}Vです"target_temperature = float(input('温度(℃)を入力してください>>'))print_temperature(target_temperature, with_celsius)print_temperature(target_temperature, with_fahrenheit)問題4def print_temperature(temperature, formatter):    print(formatter(temperature))target_temperature = float(input('温度(℃)を入力してください>>'))print_temperature(target_temperature, lambda c: f"温度は{c}℃です")print_temperature(target_temperature, lambda c: f"温度は{c * 9 / 5 + 32}Vです")
問題5class Book:    def __init__(self, title, author, price):        self.title = title        self.author = author        self.price = price    def price_with_tax(self):        return int(self.price * 1.10)    def print_info(self):        print(f'「{self.title}」（作者：{self.author}／税抜価格：{self.price}円／税込価格：{self.price_with_tax()}円）')book1 = Book('博士の愛した数式', '小川洋子', 1000)book1.print_info()book2 = Book('容疑者Xの献身', '東野圭吾', 770)book2.print_info()
問題6class Book:    total_count = 0  # クラス変数：全インスタンスの数を記録する    def __init__(self, title, author, price):        self.title = title        self.author = author        self.price = price        Book.total_count += 1  # インスタンスが作成されるたびにカウントを増やす    def price_with_tax(self):        return int(self.price * 1.10)    def print_info(self):        print(f'「{self.title}」（作者：{self.author}／税抜価格：{self.price}円／税込価格：{self.price_with_tax()}円）')    @classmethod    def print_total_count(cls):        print(f'本は{cls.total_count}冊あります')book1 = Book('博士の愛した数式', '小川洋子', 1000)book1.print_info()Book.print_total_count()book2 = Book('容疑者Xの献身', '東野圭吾', 770)book2.print_info()Book.print_total_count()
問題7class Car:    def __init__(self, maker, color, number):        self.maker = maker        self.color = color        self.number = number    def print_info(self):        print(f"メーカー: {self.maker}")        print(f"色: {self.color}")        print(f"車両番号: {self.number}")class Truck(Car):    def __init__(self, maker, color, number, capacity):        super().__init__(maker, color, number)        self.capacity = capacity    def print_info(self):        super().print_info()        print(f"最大積載量: {self.capacity}トン")mycar = Car('トヨタ', '白', '1234')mycar.print_info()  mytruck = Truck('三菱ふそう', '黒', '5678', 7)mytruck.print_info()
